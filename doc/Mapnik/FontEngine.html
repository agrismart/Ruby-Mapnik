def self.register_fonts(path, valid_extensions = DEFAULT_VALID_EXTENSIONS)
  file_names = Dir.glob(File.join("#{path}", "*.{#{valid_extensions.join(',')}}"))
  file_names.each{|filename| register_font(filename)} 
end
